    iterate_threshold::T = convert(T, 10)
    discount_factor::T = convert(T, 0.5)
    max_iters::Int = 10

    verbose::Bool = false
end

SpatialGSP.AxisSearchConfig — Type
@kwdef struct AxisSearchConfig{T <: AbstractFloat} <: GraphConstructionConfig
    kernel_σ_multiplier::T = convert(T, 3)
    make_undirected::Bool = true
    remove_null_nbs::Bool = true
    w_lb::T = convert(T, 0.1)
end

If remove_null_nbs is set to false, you can infer which neighbors came from which axis direction from the generated graph.

SpatialGSP.KNNConfig — Type
@kwdef struct KNNConfig{T <: AbstractFloat} <: GraphConstructionConfig

    normalize_min_weight::Bool = false
    make_symmetric::Bool = true
    k::Int = 4
    kernel_gain = convert(T, Inf)
    kernel_σ² = convert(T, Inf)
end
